# Kontrol edilecek dosya uzantıları
$extensions = @("*.pdf", "*.jpg", "*.jpeg", "*.png", "*.txt", "*.docx", "*.xlsx", "*.pptx")

# Aranacak dizinler
$searchPaths = @("C:\Users")

# D: sürücüsü varsa listeye ekle
if (Test-Path "D:\") {
    $searchPaths += "D:\"
}

foreach ($searchPath in $searchPaths) {
    foreach ($ext in $extensions) {
        # Dosyaları bul ve tam yolunu listele
        Get-ChildItem -Path $searchPath -Recurse -Filter $ext -ErrorAction SilentlyContinue |
        ForEach-Object { $_.FullName }
    }
}
